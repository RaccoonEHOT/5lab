128 public static bool checkDoctorData(string login, string password, string repPawssword)
129 {
130 if (login == null || login.Length == 0) { throw new Exception(ExceptionStrings.EmptyLogin); }
131 if (password == null || password.Length == 0) { throw new Exception(ExceptionStrings.EmptyPassword1); }
132 if (repPawssword == null || repPawssword.Length == 0) { throw new Exception(ExceptionStrings.EmptyPassword2); }
133 if (password != repPawssword) { throw new Exception(ExceptionStrings.DifferentPasswords); }
134 if (login == password) { throw new Exception(ExceptionStrings.SameLoginPassword); }
135 if (password.Length < 10) { throw new Exception(ExceptionStrings.PasswordLess10Chars); }
136
137 System.Text.RegularExpressions.Regex regex = new System.Text.RegularExpressions.Regex(@"[0-9]");
138
139 if (!regex.IsMatch(password)) { throw new Exception(ExceptionStrings.PasswordNoNumber); }
140
141 regex = new System.Text.RegularExpressions.Regex(@"[@#$%^&*!]");
142
143 if (!regex.IsMatch(password)) { throw new Exception(ExceptionStrings.PasswordNoExtraChar); }
144
145 regex = new System.Text.RegularExpressions.Regex(@"[A-ZА-Я]");
146
147 if (!regex.IsMatch(password)) { throw new Exception(ExceptionStrings.PasswordNoUpperChar); }
148
149 regex = new System.Text.RegularExpressions.Regex(@"^[0-9A-ZА-Яa-zа-я_]+$");
150
151 if (!regex.IsMatch(login)) { throw new Exception(ExceptionStrings.LoginForbidden); }
152
153 return true;
154 }
